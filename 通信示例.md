# 页面间通信示例

## 通信架构

```
控制面板 (index.html)
    ↕ electronAPI
主进程 (main.js)
    ↕ IPC
首页 / 其他页面 (BrowserView)
    使用 browserAPI
```

## 可用的通信 API

### 1. 其他页面 → 首页

**在任何页面（非首页）发送消息：**
```javascript
// 发送消息到首页
window.browserAPI.sendToHome({
  type: 'user-action',
  data: { userId: 123, action: 'click' }
});
```

**在首页接收消息：**
```javascript
// 监听来自其他页面的消息
window.browserAPI.onMessageFromOtherPage((message) => {
  console.log('收到消息:', message);
  // message = { type: 'user-action', data: { userId: 123, action: 'click' } }
});
```

### 2. 首页 → 其他页面

**在首页发送消息：**
```javascript
// 从首页发送消息到其他页面
window.browserAPI.sendToOtherPage({
  type: 'config-update',
  config: { theme: 'dark', language: 'zh-CN' }
});
```

**在其他页面接收消息：**
```javascript
// 监听来自首页的消息
window.browserAPI.onMessageFromHome((message) => {
  console.log('收到首页消息:', message);
  // message = { type: 'config-update', config: { theme: 'dark', language: 'zh-CN' } }

  // 应用配置
  if (message.type === 'config-update') {
    applyConfig(message.config);
  }
});
```

### 3. 控制面板 → 任何页面

**在控制面板 (renderer.js) 发送：**
```javascript
// 向当前显示的页面发送消息
window.electronAPI.sendToContent({
  command: 'highlight',
  selector: '#main-content'
});
```

**在内容页面接收：**
```javascript
// 监听来自控制面板的消息
window.browserAPI.onMessageFromMain((message) => {
  console.log('收到控制面板消息:', message);

  if (message.command === 'highlight') {
    document.querySelector(message.selector).style.border = '2px solid red';
  }
});
```

## 完整示例

### 示例 1：首页配置分发

**首页代码 (http://localhost:5173/)：**
```javascript
// 在首页加载时初始化
window.addEventListener('DOMContentLoaded', () => {
  // 监听其他页面的请求
  window.browserAPI.onMessageFromOtherPage((message) => {
    if (message.type === 'request-config') {
      // 发送配置给其他页面
      window.browserAPI.sendToOtherPage({
        type: 'config',
        data: {
          apiKey: 'your-api-key',
          baseURL: 'https://api.example.com',
          theme: 'dark'
        }
      });
    }
  });

  // 当配置改变时，主动推送
  function onConfigChange(newConfig) {
    window.browserAPI.sendToOtherPage({
      type: 'config-update',
      data: newConfig
    });
  }
});
```

**其他页面代码（通过脚本注入）：**
```javascript
// 在页面加载时请求配置
window.addEventListener('DOMContentLoaded', () => {
  // 监听首页的配置消息
  window.browserAPI.onMessageFromHome((message) => {
    if (message.type === 'config') {
      console.log('收到配置:', message.data);
      // 使用配置
      localStorage.setItem('config', JSON.stringify(message.data));
    }

    if (message.type === 'config-update') {
      console.log('配置已更新:', message.data);
      // 更新本地配置
      localStorage.setItem('config', JSON.stringify(message.data));
    }
  });

  // 向首页请求配置
  window.browserAPI.sendToHome({ type: 'request-config' });
});
```

### 示例 2：页面状态同步

**在任何页面发送状态到首页：**
```javascript
// 用户登录后
function onUserLogin(userData) {
  window.browserAPI.sendToHome({
    type: 'user-login',
    user: {
      id: userData.id,
      name: userData.name,
      token: userData.token
    }
  });
}

// 用户执行操作
document.getElementById('submit-btn').addEventListener('click', () => {
  window.browserAPI.sendToHome({
    type: 'user-action',
    action: 'submit-form',
    timestamp: Date.now()
  });
});
```

**首页接收并处理：**
```javascript
window.browserAPI.onMessageFromOtherPage((message) => {
  switch (message.type) {
    case 'user-login':
      // 保存用户信息
      saveUserSession(message.user);

      // 通知其他可能打开的页面
      window.browserAPI.sendToOtherPage({
        type: 'session-update',
        user: message.user
      });
      break;

    case 'user-action':
      // 记录用户行为
      logUserAction(message.action, message.timestamp);
      break;
  }
});
```

### 示例 3：脚本注入自动通信

**在脚本注入面板注入以下代码：**
```javascript
// 页面加载完成后自动与首页建立连接
console.log('页面已注入通信脚本');

// 监听首页消息
window.browserAPI.onMessageFromHome((message) => {
  console.log('收到首页消息:', message);

  // 根据消息类型执行不同操作
  if (message.type === 'auto-fill') {
    // 自动填充表单
    document.getElementById('username').value = message.username;
    document.getElementById('password').value = message.password;
  }
});

// 页面就绪后通知首页
window.browserAPI.sendToHome({
  type: 'page-ready',
  url: window.location.href,
  title: document.title
});

// 监听表单提交
document.querySelector('form')?.addEventListener('submit', (e) => {
  // 通知首页表单已提交
  window.browserAPI.sendToHome({
    type: 'form-submitted',
    url: window.location.href
  });
});
```

## 注意事项

1. **消息只发送到当前显示的页面**：通信是实时的，消息不会被缓存
2. **首页判断**：系统通过 URL `http://localhost:5173/` 判断是否为首页
3. **消息格式**：建议使用 `{ type: 'xxx', data: {...} }` 格式便于处理
4. **错误处理**：所有通信都有错误捕获，失败不会影响页面运行
5. **异步通信**：消息发送是异步的，不要期待立即收到响应

## 调试技巧

在任何页面的控制台中：
```javascript
// 查看是否有 browserAPI
console.log(window.browserAPI);

// 测试发送消息到首页
window.browserAPI.sendToHome({ test: 'hello' });

// 测试从首页发送消息（在首页执行）
window.browserAPI.sendToOtherPage({ test: 'hello from home' });

// 添加通用消息监听器
window.addEventListener('message', (event) => {
  console.log('收到 postMessage:', event.data);
});
```
