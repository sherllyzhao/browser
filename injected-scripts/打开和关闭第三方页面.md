# 打开和关闭第三方页面

## 概述

在注入脚本中，你可以使用 `window.open()` 打开第三方页面（如抖音），并在完成任务后关闭该页面。

## 基本用法

### 1. 打开第三方页面

```javascript
// 全局变量，保存打开的窗口引用
let douyinWindow = null;

// 打开抖音页面的函数
function openDouyinPage() {
  // 打开新窗口，保存引用
  douyinWindow = window.open('https://www.douyin.com', '_blank');

  console.log('✅ 已打开抖音页面');

  return douyinWindow;
}
```

### 2. 关闭第三方页面

```javascript
// 关闭抖音页面的函数
function closeDouyinPage() {
  if (douyinWindow && !douyinWindow.closed) {
    douyinWindow.close();
    console.log('✅ 已关闭抖音页面');
    douyinWindow = null;
    return true;
  } else {
    console.log('⚠️ 没有打开的抖音窗口可关闭');
    return false;
  }
}
```

### 3. 监听窗口是否被关闭

```javascript
function openDouyinPage() {
  douyinWindow = window.open('https://www.douyin.com', '_blank');

  // 监听窗口是否被手动关闭
  const checkWindowClosed = setInterval(() => {
    if (douyinWindow && douyinWindow.closed) {
      console.log('ℹ️ 抖音窗口已被关闭');
      clearInterval(checkWindowClosed);
      douyinWindow = null;
    }
  }, 1000);

  return douyinWindow;
}
```

```json
    "http://localhost:5173/#/short-video/auth": "auth.js",
    "https://www.douyin.com/*": "douyin-creator.js"
  }
```


## 父子窗口通信

### 从父窗口（auth）向子窗口（douyin）发送消息

在 `auth.js` 中：

```javascript
// 打开抖音页面
function openDouyinPage() {
  douyinWindow = window.open('https://www.douyin.com', '_blank');
  console.log('✅ 已打开抖音页面');

  // 等待窗口加载完成后发送消息
  setTimeout(() => {
    sendMessageToDouyin({
      type: 'START_TASK',
      data: { userId: 123, action: 'auto-like' }
    });
  }, 3000); // 等待3秒让页面加载

  return douyinWindow;
}

// 向抖音窗口发送消息
function sendMessageToDouyin(message) {
  if (douyinWindow && !douyinWindow.closed) {
    douyinWindow.postMessage(message, 'https://www.douyin.com');
    console.log('✅ 已发送消息到抖音:', message);
    return true;
  } else {
    console.log('⚠️ 抖音窗口未打开或已关闭');
    return false;
  }
}

// 监听来自抖音窗口的回复
window.addEventListener('message', (event) => {
  // 安全检查：验证消息来源
  if (event.origin !== 'https://www.douyin.com') {
    return;
  }

  console.log('✅ 收到抖音消息:', event.data);

  if (event.data.type === 'TASK_COMPLETE') {
    console.log('任务完成，准备关闭窗口');
    setTimeout(() => closeDouyinPage(), 1000);
  }
});
```

### 在子窗口（douyin）接收消息并回复

在 `douyin-creator.js` 中：

```javascript
// 监听来自父窗口的消息
window.addEventListener('message', (event) => {

```

### 完整通信示例

#### auth.js (父窗口)
```javascript
let douyinWindow = null;

// 1. 打开抖音页面
function openDouyinPage(taskConfig) {
  douyinWindow = window.open('https://www.douyin.com', '_blank');

  // 等待页面加载后发送配置
  setTimeout(() => {
    sendMessageToDouyin({
      type: 'INIT_TASK',
      config: taskConfig
    });
  }, 3000);

  return douyinWindow;
}

// 2. 发送消息到抖音
function sendMessageToDouyin(message) {
  if (douyinWindow && !douyinWindow.closed) {
    douyinWindow.postMessage(message, 'https://www.douyin.com');
    console.log('📤 发送消息到抖音:', message);
    return true;
  }
  console.warn('⚠️ 抖音窗口未打开');
  return false;
}

// 3. 接收来自抖音的消息
window.addEventListener('message', (event) => {
  // 验证来源
  if (event.origin !== 'https://www.douyin.com') return;

  console.log('📥 收到抖音消息:', event.data);

  switch (event.data.type) {
    case 'READY':
      // 抖音页面已准备好，可以发送指令
      sendMessageToDouyin({
        type: 'START_TASK',
        data: { action: 'auto-like', count: 10 }
      });
      break;

    case 'PROGRESS':
      // 任务进度更新
      console.log(`任务进度: ${event.data.progress}%`);
      break;

    case 'COMPLETE':
      // 任务完成
      console.log('✅ 任务完成:', event.data.result);
      setTimeout(() => closeDouyinPage(), 1000);
      break;

    case 'ERROR':
      // 任务失败
      console.error('❌ 任务失败:', event.data.error);
      break;
  }
});

// 4. 关闭抖音窗口
function closeDouyinPage() {
  if (douyinWindow && !douyinWindow.closed) {
    douyinWindow.close();
    douyinWindow = null;
    console.log('🔒 已关闭抖音窗口');
  }
}

// 示例：点击按钮打开抖音并执行任务
document.getElementById('startBtn')?.addEventListener('click', () => {
  openDouyinPage({
    userId: 123,
    taskType: 'auto-like',
    count: 10
  });
});
```

#### douyin-creator.js (子窗口)
```javascript
console.log('✅ douyin-creator.js 已注入');

(function() {
  let taskConfig = null;

  // 1. 监听父窗口消息
  window.addEventListener('message', async (event) => {
    // 验证来源（根据实际情况修改域名）
    if (event.origin !== 'http://localhost:5173') return;

    console.log('📥 收到指令:', event.data);

    switch (event.data.type) {
      case 'INIT_TASK':
        // 接收任务配置
        taskConfig = event.data.config;
        notifyParent('READY', { message: '已准备就绪' });
        break;

      case 'START_TASK':
        // 开始执行任务
        await executeTask(event.data.data);
        break;

      case 'STOP_TASK':
        // 停止任务
        console.log('停止任务');
        break;
    }
  });

  // 2. 向父窗口发送消息
  function notifyParent(type, data) {
    if (window.opener && !window.opener.closed) {
      window.opener.postMessage(
        { type, ...data },
        'http://localhost:5173'
      );
      console.log('📤 通知父窗口:', type, data);
    }
  }

  // 3. 执行自动化任务
  async function executeTask(taskData) {
    try {
      const { action, count } = taskData;
      console.log(`开始执行: ${action}, 数量: ${count}`);

      for (let i = 0; i < count; i++) {
        // 执行具体操作
        await performAction(action);

        // 报告进度
        const progress = Math.round(((i + 1) / count) * 100);
        notifyParent('PROGRESS', { progress });

        // 等待一下
        await sleep(1000);
      }

      // 任务完成
      notifyParent('COMPLETE', {
        result: `成功完成 ${count} 次 ${action}`
      });

      // 3秒后自动关闭
      setTimeout(() => window.close(), 3000);

    } catch (error) {
      notifyParent('ERROR', { error: error.message });
    }
  }

  // 4. 执行具体操作
  async function performAction(action) {
    switch (action) {
      case 'auto-like':
        // 点赞逻辑
        console.log('执行点赞...');
        // 找到点赞按钮并点击
        const likeBtn = document.querySelector('.like-button');
        if (likeBtn) likeBtn.click();
        break;

      case 'auto-follow':
        // 关注逻辑
        console.log('执行关注...');
        break;

      // 其他操作...
    }
  }

  // 工具函数
  function sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  // 页面加载完成后通知父窗口
  window.addEventListener('load', () => {
    notifyParent('READY', { message: '页面加载完成' });
  });

})();
```

### 常用消息类型

```javascript
{

  type: 'PROGRESS',       // 进度更新

  type: 'ERROR',          // 发生错误
```

## 安全注意事项

1. **验证消息来源**: 始终检查 `event.origin`
```javascript
if (event.origin !== 'https://expected-domain.com') {
  return; // 忽略未知来源的消息
}
```

2. **指定目标域名**: 使用 `postMessage` 时指定目标
```javascript
// ✅ 好的做法
window.opener.postMessage(data, 'https://specific-domain.com');

// ❌ 不安全的做法
window.opener.postMessage(data, '*'); // 任何网站都能接收
```

3. **验证消息内容**: 检查消息格式和内容
```javascript
if (!event.data || typeof event.data.type !== 'string') {
  return; // 忽略格式错误的消息
}
```

## 注意事项

1. **窗口引用**: 必须保存 `window.open()` 的返回值才能后续关闭窗口
2. **跨域限制**: 第三方网站无法直接操作（除了关闭窗口），但可以通过注入脚本来控制
3. **弹窗拦截**: 浏览器可能会拦截 `window.open()`，确保在用户操作（如点击按钮）时调用
4. **自动关闭**: 使用 `window.close()` 可以关闭由脚本打开的窗口
5. **消息通信**: 使用 `postMessage` 在父子窗口间传递数据

## 工作流程

1. **主页面** (auth.js) 调用 `window.open()` 打开抖音
2. **抖音页面**加载完成后，自动注入 `douyin-creator.js`
3. **douyin-creator.js** 执行自动化任务
4. 任务完成后，通过 `postMessage` 通知父窗口
5. 父窗口收到消息后，调用 `window.close()` 关闭抖音窗口

## 实用工具函数

### 等待元素出现

```javascript
function waitForElement(selector, timeout = 10000) {
  return new Promise((resolve, reject) => {
    const element = document.querySelector(selector);
    if (element) {
      resolve(element);
      return;
    }

    const observer = new MutationObserver(() => {
      const element = document.querySelector(selector);
      if (element) {
        observer.disconnect();
        resolve(element);
      }
    });

    observer.observe(document.body, {
      childList: true,
      subtree: true
    });

    setTimeout(() => {
      observer.disconnect();
      reject(new Error(`元素 ${selector} 未在 ${timeout}ms 内找到`));
    }, timeout);
  });
}

// 使用示例
const button = await waitForElement('.some-button');
button.click();
```
